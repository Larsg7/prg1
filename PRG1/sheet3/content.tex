\subsubsection*{$1234_{10}$}
\begin{enumerate}
	\item \textbf{Basis 2:}\\
	$1234 / 2 = 617 \text{R} 0$\\
	$617 / 2 = 308 \text{R} 1$\\
	$308 / 2 = 154 \text{R} 0$\\ 
	$154 / 2 = 77 \text{R} 0$\\
	$77 / 2 = 38 \text{R} 1$\\
	$38 / 2 = 19 \text{R} 0$\\
	$19 / 2 = 9 \text{R} 1$\\
	$9 / 2 = 4 \text{R} 1$\\
	$4 / 2 = 2 \text{R} 0$\\
	$2 / 2 = 1 \text{R} 0$\\
	$1 / 2 = 0 \text{R} 1$\\
	$\rightarrow 1234_{10} = 0000\ 0100\ 1101\ 0010_2$
	\item \textbf{Basis 16:}\\
	Von Bin√§r nach Hexadecimal:\\
	$0010_2 = 2_{16}$\\
	$1101_2 = D_{16}$\\
	$0100_2 = 4_{16}$\\
	$\rightarrow 1234_{10} = 4D2_{16}$
\end{enumerate}
